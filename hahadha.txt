2. Skaičius egzempliorių, kurie buvo gauti bibliotekoje praėjusį mėnesį ir jau buvo paimti skaityti.

SELECT COUNT(DISTINCT e.nr) AS borrowed_received_last_month
FROM Stud.Egzempliorius AS e
JOIN Stud.Skaitymas AS s ON s.egzempliorius = e.nr
WHERE e.gavimo_data >= date_trunc('month', CURRENT_DATE) - INTERVAL '1 month'
  AND e.gavimo_data <  date_trunc('month', CURRENT_DATE)
  AND s.paimta IS NOT NULL;
