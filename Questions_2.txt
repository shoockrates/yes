#include <stdio.h>

#define TRUE (1==1)
#define FALSE (!TRUE)

typedef struct tagPOINT {
    int x;
    int y;
} POINT;

int returnSquare(int num) {
    return num * num;
}

int isRightTriangle(POINT A, POINT B, POINT C) {
    int AB_2 = returnSquare(A.x - B.x) + returnSquare(A.y - B.y);
    int BC_2 = returnSquare(B.x - C.x) + returnSquare(B.y - C.y);
    int CA_2 = returnSquare(C.x - A.x) + returnSquare(C.y - A.y);

    if (AB_2 + BC_2 == CA_2 || AB_2 + CA_2 == BC_2 || CA_2 + BC_2 == AB_2) {
        return TRUE;
    } else {
        return FALSE;
    }
}
